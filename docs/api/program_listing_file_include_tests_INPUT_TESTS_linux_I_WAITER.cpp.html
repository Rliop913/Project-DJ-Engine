void
futex_wait(int* addr)
{
    syscall(SYS_futex, addr, FUTEX_WAIT, 0, NULL, NULL, 0);
}

void
futex_wake(int* addr)
{
    syscall(SYS_futex, addr, FUTEX_WAKE, 1, NULL, NULL, 0);
}
