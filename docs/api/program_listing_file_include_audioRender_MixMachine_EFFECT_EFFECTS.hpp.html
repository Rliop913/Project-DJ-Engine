template<typename Fclass>
class PDJE_API FaustDType : public Fclass{
private:
    std::vector<float> L;
    std::vector<float> R;
    SIMD_FLOAT* originVec;
    unsigned long long startPos;
public:
    int count;
    FAUSTFLOAT* PTR[CHANNEL];

    FaustDType(
        SIMD_FLOAT* vec,
        unsigned long long start,
        unsigned long long end):
        startPos(start),
        originVec(vec)
    {
        count = end - start;
        L.resize(count);
        R.resize(count);
        PTR[0] = L.data();
        PTR[1] = R.data();
    }
    void CopyToFaust(){
        float* lp = L.data();
        float* rp = R.data();
        float* op = originVec->data();
        op += startPos * CHANNEL;
        for(int i=0; i<count; ++i){
            *(lp++) = *(op++);
            *(rp++) = *(op++);
        }
    }
    void WriteToOrigin()
    {
        float* lp = L.data();
        float* rp = R.data();
        float* op = originVec->data();
        op+= startPos * CHANNEL;
        for(int i=0; i<count; ++i){
            *(op++) = *(lp++);
            *(op++) = *(rp++);
        }
    }

};

template<typename FaustClass>
class FaustObject{

public:
    FaustClass managingClass;

    template<typename FClass>
    void consume(std::vector<FaustDType<FClass>>& jobs)
    {
        for(auto& i : jobs){
            i.CopyToFaust();
            i.template copySetting<FClass>(managingClass);
            managingClass.instanceClear();
            managingClass.compute(i.count, i.PTR, i.PTR);
            i.WriteToOrigin();
        }
    }
};
