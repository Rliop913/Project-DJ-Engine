inline
void
toFaustStylePCM(float** faustPCM, float* in, const unsigned long frameCount)
{
    float* op = in;
    float* lp = faustPCM[0];
    float* rp = faustPCM[1];
    for(int i=0; i<frameCount; ++i){
        *(lp++) = *(op++);
        *(rp++) = *(op++);
    }
}
